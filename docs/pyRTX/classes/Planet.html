356	def compute(self, epochs, step):
357		"""
358		Implement a compute method to calculate the temperature grid in the sun-fixed frame
359		(X-axis pointing the Sun).
360  
361		Input: 
362  		- list of epochs
363		- Latitude/Longitude step in degree
364
365		"""
366  
367		# -------------------------------------------------------------------
368		# COMPUTE SUN DIRECTION
369
370		sunVec = sp.spkezr('Sun', np.array(epochs), self._frame, 'None', self._planet_name)
371		sunVec = [sunVec[0][i][0:3] for i in range(len(epochs))]
372
373		# -------------------------------------------------------------------
374		# DEFINE MESH GRID
375
376		# Compute grid of latitude and Lonitude (DO NOT CHANGE RESOLUTION)
377		Lon      = np.arange(-180, 181, step)
378		Lat      = np.arange(90, -91, -step)
379
380		# Number of rows and columns
381		nrows = len(Lon)
382		ncols = len(Lat)
383  
384		# Init array
385		self._data  = np.zeros((len(epochs), nrows, ncols))
386   
387		for i, sunvec in enumerate(sunVec):
388		
389			for r in range(nrows):
390				
391				for c in range(ncols):
392        
393					lon = Lon[r]*np.pi/180.
394					lat = Lat[c]*np.pi/180.
395
396					# Cell normal unit vector
397					nij  = np.array( [np.cos(lat)*np.cos(lon),
398									  np.cos(lat)*np.sin(lon),
399									  np.sin(lat),] )
400		
401					# Get albedo value
402					if isinstance(self._albedo, AlbedoGrid):
403						alb = self.get_albedo(epochs[i], nij)
404					elif isinstance(self._albedo, float):
405						alb = self._albedo
406
407					# Get emissivity value
408					if isinstance(self._emissivity, EmissivityGrid):
409						emi = self.get_emissivity(epochs[i], nij)
410					elif isinstance(self._emissivity, float):
411						emi = self._emissivity
412     
413					# Sun directions
414					cellSunDir = [-self._radius*nij[0] + sunvec[0],  
415								  -self._radius*nij[1] + sunvec[1],  
416								  -self._radius*nij[2] + sunvec[2]]
417
418					# Scaled solar flux
419					phi = self._base_flux*((au/np.linalg.norm(cellSunDir))**2)
420					
421					# Sun-aspect angle
422					x          = np.dot(nij, cellSunDir/np.linalg.norm(cellSunDir))
423					GammaAngle = np.arccos(x)
424					inLight    = -np.pi/2 < GammaAngle < np.pi/2
425     
426					# Compute temperature				
427					if inLight:      
428						
429						# In-light condition
430						self._data[i,r,c]  =  ((1-alb) * phi * np.cos(GammaAngle) / (sb*emi))**0.25
431     
432					else: 
433						
434						# Terminator condition
435						self._data[i,r,c] = self._nightside_temp
436      
437					if self._data[i,r,c] < self._nightside_temp: self._data[i,r,c] = self._nightside_temp
438     
439		# Build the x-array dataset
440		self._data = xr.Dataset(
441						data_vars=dict(
442							temperature=(["time", "lon", "lat"], self._data),
443						),
444						coords=dict(
445							lon = Lon,
446							lat = Lat,
447							time = epochs,
448						),
449						attrs=dict(frame=self._frame,),
450					)
451
452		# Store dimensions and axes
453		self._dims  = self._data.temperature.dims
454		self._axes  = [self._data.coords[key].data for key in list(self._dims)]
