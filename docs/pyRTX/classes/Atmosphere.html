176	def compute(self, epoch, lat, lon, height, inputUnits = 'deg', variabilities = None):
177
178
179		if isinstance(epoch, float):
180			epoch = [epoch]
181		if isinstance(lat, float):
182			lat = [lat]
183		if isinstance(lon, float):
184			lon = [lon]
185		if isinstance(height, float):
186			height = [height]
187
188
189
190		#Check length of arrays
191		if len(lat) != len(lon):
192			raise ValueError('Latitude and Longitude arrays must be of same length')
193		if len(lat) != len(epoch):
194			raise ValueError('The epoch array must be of same lehgth of lat and lon')
195		if len(height) != len(epoch):
196			raise ValueError('The height array must be of same lehgth of lat and lon')
197
198		if inputUnits == 'rad':
199			conv = np.pi/180.0
200			lat = lat * conv
201			lon = lon * conv
202	
203
204		rho = np.zeros(len(epoch))
205		T = np.zeros(len(epoch))
206		P = np.zeros(len(epoch))
207		composition = np.zeros((len(epoch), 7))
208		if variabilities != None:
209			var_low = np.zeros(len(epoch))
210			var_avg = np.zeros(len(epoch))
211			var_hig = np.zeros(len(epoch))
212
213		for i, ep in enumerate(epoch):
214			self.namelistWriter(ep, lat[i], lon[i], height[i])
215			os.system( 'echo {} | {}venusgrm_V05.x 2&> {}/log.txt'.format(self.tmpFolder + '/input.txt', self.execFolder, self.tmpFolder))
216			rho[i], T[i], P[i], composition[i,:] = self.readResults()
217			if variabilities != None:
218				var_low[i], var_avg[i], var_hig[i] = self.readVariabilities(variabilities)
219
220				
221
222		if variabilities == None:
223			return rho, T, P,composition
224		else:
225			return rho, T, P, composition, var_low, var_avg, var_hig
