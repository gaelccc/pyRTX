Compute Lookup Table

This example demonstrates lookup table generation when the spacecraft has moveable appendages

compute_lut.py
  1### -------------------------------------------------------------------- ###
  2
  3#					    LOOK UP TABLE OBJECT BUILDING
  4
  5# Example case:
  6# Generate the normalized optical response lookup table for LRO.
  7
  8# The look up table is designed to store the "geometry vector" of a spacecraft.
  9# This vector is representative of how the spacecraft’s shape and surface 
 10# properties interact with radiation coming from a specific direction. 
 11
 12# The lookup table helps to speed up the computation of non-gravitational
 13# accelerations and is MANDATORY for albedo, thermal infrared and drag
 14# acceleration.
 15
 16# To compute a lookup table, we need to define a grid of right ascensions
 17# and declinations that represents the directions from which the radiation 
 18# is coming in the spacecraft body-fixed frame. 
 19
 20# Also, we need to specify every frame that is not fixed with respect to 
 21# the body-fixed frame. These frames are stored in the variable 'moving_frames'. 
 22# If a frame is not specified inside 'moving_frames', it is represented 
 23# fixed with respect to the spacecraft body frame. For every moving_frame, 
 24# a range of euler angles in a specific euler set must be defined.
 25
 26# The LUT have two computational mode:
 27# - If units is 'km/s**2', the LUT computes the acceleration for a 
 28#   normalized radiation flux. The LUT values will be 3x1 vectors. This
 29#	mode is for srp, albedo and thermal infrared acceleration.
 30# - If units is 'km**2', the LUT computes the cross-section for the
 31#	drag acceleration. The LUT values will be single float. 
 32
 33# NOTE: Here we compute the lookup table for the Lunar Redonnaissance orbiter
 34# by varying the orientation of the solar array. The High Gain Antenna
 35# is considered fixed with respect to the bus. 
 36
 37### -------------------------------------------------------------------- ###
 38### IMPORTS
 39
 40import time
 41import xarray as xr
 42import spiceypy as sp
 43import numpy as np
 44import pickle as pkl
 45
 46from numpy import floor, mod
 47from math import ceil
 48
 49from pyRTX.classes.SRP import SolarPressure
 50from pyRTX.classes.Spacecraft import Spacecraft
 51from pyRTX.classes.PixelPlane import PixelPlane
 52from pyRTX.classes.RayTracer import RayTracer
 53from pyRTX.core.analysis_utils import epochRange2
 54
 55import multiprocessing as mproc
 56
 57import timeit, os, itertools, logging
 58
 59### -------------------------------------------------------------------- ###
 60### INPUTS & LUT CONFIG
 61			  	
 62n_cores    =  10							      # number of cores for parallel computation
 63grid_res   =  20							      # angular resolution for RA, DEC
 64angle_res  =  10						          # angular resolution for moving parts
 65spacing	   =  0.01								  # spacing between rays 
 66
 67ref_epc	    = "2010 may 10 09:25:00"	          # reference epoch
 68duration    = 10000  							  # seconds
 69timestep    = 100
 70
 71sc_mass 	  = 1								  # the sc mass must be 1 for LUT computation
 72base_frame    = 'LRO_SC_BUS'				      # sc body-fixed frame
 73moving_frames = ['LRO_SA',]					      # frames that are not fixed wrt the sc base frame
 74
 75eul_set = (2,1,3)								  # euler representation for moving frames attitude
 76
 77obj_path = '../example_data/LRO/'		          # path for 3D shape elements
 78
 79METAKR = '../example_data/LRO/metakernel_lro.tm'  # metakernel
 80
 81units  = 'km/s**2'	                              # units for lookup table values: km/s**2 or km**2
 82
 83lutfile =  'luts/lro_accel_lut.nc'	    		  # output file
 84
 85### --------------------------------------------------------------------------- ###
 86### LUT LIMITS
 87
 88sp.furnsh(METAKR)
 89
 90# List of right ascension and declination values for the incoming rays
 91RA   = np.linspace(0, 360, int(360/grid_res) + 1) * np.pi / 180
 92DEC  = np.linspace(-90, 90, int(180/grid_res) + 1) * np.pi / 180
 93
 94###  Here we find the limits for every moving frame, in terms of euler angles  ###
 95
 96# Define timespan
 97eul_limits = {frame: {e: [] for e in eul_set} for frame in moving_frames}
 98epc_et0  = sp.str2et( ref_epc )
 99epc_et1  = epc_et0 + duration
100epochs   = epochRange2(startEpoch = epc_et0, endEpoch = epc_et1, step = timestep)
101
102# Find euler limits
103for frame in moving_frames:
104    
105	EUL = np.zeros((len(epochs),3))
106
107	for i, epc in enumerate(epochs):
108
109		rot = sp.pxform(frame, base_frame, epc)
110		EUL[i,:] = np.array(sp.m2eul(rot, *eul_set)) * 180 / np.pi
111 
112	for i, e in enumerate(eul_set):
113  
114		if max(abs(EUL[:,i].max()), abs(EUL[:,i].min())) >= grid_res:
115			eul_limits[frame][e] = [EUL[:,i].min() - 0.05, EUL[:,i].max() + 0.05]
116
117sp.unload(METAKR)
118
119### --------------------------------------------------------------------------- ###
120### OBJECT DEFINITION
121
122# Define a spacecraft model
123spacecraft_model = {						        
124                          
125					'LRO_BUS': { 
126							 'file' : obj_path + 'bus_rotated.obj',	 # .obj file of the spacecraft component
127							 'frame_type': 'Spice',				     # type of frame (can be 'Spice' or 'UD'
128							 'frame_name': 'LRO_SC_BUS',			 # Name of the frame
129							 'center': [0.0,0.0,0.0],			     # Origin of the component
130							 'diffuse': 0.1,				         # Diffuse reflect. coefficient
131							 'specular': 0.3,				         # Specular reflect. coefficient
132							 },
133
134					'LRO_SA': {	
135							'file': obj_path + 'SA_recentred.obj',
136							'frame_type': 'Spice',
137							'frame_name': 'LRO_SA',
138							'center': [-1,-1.1, -0.1],
139							'diffuse': 0,
140							'specular': 0.3,
141							},
142
143
144					'LRO_HGA': { 	
145							'file': obj_path + 'HGA_recentred.obj',
146							'frame_type': 'Spice',
147							'frame_name': 'LRO_HGA',
148							'center':[-0.99,    -0.3,  -3.1],
149							'diffuse': 0.2,
150							'specular': 0.1,
151							},
152					}
153 
154 # Elements with moving frames must have an user defined rotation
155for elem in spacecraft_model.keys():
156    if any([spacecraft_model[elem]['frame_name'] == frame for frame in moving_frames]):
157        spacecraft_model[elem]['frame_type'] = 'UD'
158        spacecraft_model[elem]['UD_rotation'] = np.identity(4)
159
160# Define the Spacecraft Object (Refer to the class documentation for further details)
161lro = Spacecraft( name = 'LRO',     
162				  base_frame = 'LRO_SC_BUS', 					     # Name of the spacecraft body-fixed frame
163                  mass = sc_mass,									 # The mass should be 1 for LUT computation 
164				  spacecraft_model = spacecraft_model,
165				)
166
167# Define the Sun rays object
168rays = PixelPlane( spacecraft = lro,
169				   mode = 'Fixed',
170				   width = 15,
171				   height = 15,
172				   ray_spacing = spacing, 
173				   lon = 0,
174				   lat = 0,
175				   distance = 30)
176
177# Define the Ray Tracer
178rtx = RayTracer( lro,                    # Spacecraft object
179				 rays,                   # pixelPlane object
180				 kernel = 'Embree3',     # The RTX kernel to use (use Embree 3)
181				 bounces = 1,            # The number of bounces to account for
182				 diffusion = False,      # Account for secondary diffusion
183				 ) 
184
185# Define the Solar Pressure Object 
186# NOTE: for LUT computation the baseflux must be set to None.
187srp = SolarPressure( lro, rtx, baseflux = None, )
188
189### -------------------------------------------------------------------- ###
190### LUT INITIALIZATION
191
192# Time initialization
193tic = timeit.default_timer()
194
195# Refresh inputs and output directiories
196os.system('rm inputs/*')
197os.system('rm outputs/*')
198
199# Save srp object
200with open('inputs/srp.pkl', 'wb') as f: pkl.dump(srp, f)
201
202# Deactivate trimesh logging
203log = logging.getLogger('trimesh')
204log.disabled = True
205
206print('\n *** Calculating dimension ...')
207
208# Build dimensions and axes
209axes = []
210dims = []
211for name in moving_frames:
212    for ax in eul_set:
213        
214        if not len(eul_limits[name][ax]): continue
215        lb = eul_limits[name][ax][0]
216        ub = eul_limits[name][ax][1]
217        
218        dims.append('%s%d'%(name,ax))
219        axes.append(np.linspace(lb, ub, ceil((ub-lb)/angle_res) + 1)*np.pi/180)	  
220
221# Append ra and dec
222dims.append('ra')
223dims.append('dec')
224dims.append('value')
225axes.append(RA) 
226axes.append(DEC) 
227
228# Build attribute dictionary for xarray
229attrs = {
230	'moving_frames': ",".join(moving_frames),
231	'base_frame': base_frame,
232	'units': units,
233	'ref_epoch': ref_epc,
234	'eul_set': ",".join([str(e) for e in eul_set]),
235	'dims': ",".join(dims),
236	}
237
238# Save attributes dictionary
239with open('inputs/attrs.pkl', 'wb') as f: pkl.dump(attrs, f)
240
241# Compute shape for xarray
242shape = tuple([len(r) for r in axes] + [3])
243
244# Build coordinates
245coords = {dims[i]: vals for i, vals in enumerate(axes)}
246
247### -------------------------------------------------------------------- ###
248### LUT PARALLEL COMPUTATION 
249
250print(f'\n *** LUT size: {shape} ...')
251
252# Init data array
253data  = np.zeros(shape)
254
255# Find all permutations
256SEQ    = list(itertools.product(*axes))
257IDX    = list(itertools.product(*[range(l) for l in shape[:-1]]))
258steps  = [int(i) for i in np.linspace(0,len(SEQ),n_cores)]
259SEQ    = [SEQ[i:j] for (i,j) in zip(steps[:-1],steps[1:])]
260IDX    = [IDX[i:j] for (i,j) in zip(steps[:-1],steps[1:])]
261INPUTS = [(I,S) for I,S in zip(IDX,SEQ)]
262
263# Save inputs file
264with open('inputs/inputs.pkl', 'wb') as f: pkl.dump(INPUTS, f)
265
266### -------------------------------------------------- ###
267### Multiprocessing target function
268
269def process(ID, METAKR):
270	os.system(f'python task_lut.py {ID} {METAKR}')
271	return
272
273### -------------------------------------------------- ###
274
275print('\n *** Filling the LUT values ...')
276
277# --------------------------------- #
278# PYTHON MULTIPROCESSING
279
280p = [0.] * len(INPUTS)
281
282# Process in parallel
283for ID in range(len(INPUTS)):
284	p[ID] = mproc.Process( target=process, args=(ID, METAKR,) )
285
286for ID in range(len(INPUTS)):       
287	p[ID].start()
288	while sum([pi.is_alive() for pi in p]) >= n_cores: time.sleep(0.5)
289
290for ID in range(len(INPUTS)): 
291	while(p[ID].is_alive()): p[ID].join(1)
292
293# --------------------------------- #
294
295# Fill LUT values
296for ID in range(len(INPUTS)):
297    result = np.load(f'outputs/output{ID}.npy')
298    for i, idxs in enumerate(INPUTS[ID][0]):
299        data[tuple(idxs)] = result[i]
300
301print('\n *** Generating the x-array ...')
302
303# Define X-array LUT
304data = xr.Dataset( data_vars = {'look_up_table': (dims, data)}, 
305				   coords    = coords,
306				   attrs     = attrs,)
307
308print(f'\n *** LUT completed!\n')
309
310# Reactivate trimesh logging
311log.disabled = False
312
313# Save
314data.to_netcdf(lutfile, encoding = data.encoding.update({'zlib': True, 'complevel': 1}))
315
316### ... Elapsed time
317toc = timeit.default_timer()
318time_min = int(floor((toc-tic)/60))
319time_sec = int(mod((toc-tic), 60))
320print("")
321print("\t Elapsed time: %d min, %d sec" %(time_min, time_sec))
322print("")
323
