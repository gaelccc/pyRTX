LRO Albedo and Thermal IR (Simple)

This example demonstrates how to compute albedo and thermal infrared accelerations using the object-oriented interface of the pyRTX library.

In this simplified case, we use single uniform values for:

  • Planet emissivity

  • Planet albedo

  • Planet surface temperature

This provides a quick way to compute planetary radiation pressure effects without requiring detailed surface property maps.

lro_alb_ir_simple.py
  1### ------------------------------------------------------------------------------------------------------- ###
  2
  3# Example purpose:
  4# Show the object-oriented interface of the pyRTX library
  5#
  6# Example case:
  7# Show how to compute albedo and thermal-ir accelerations
  8# Here we use a single value for the emissivity, albedo and temperature for the planet.
  9
 10### ------------------------------------------------------------------------------------------------------- ###
 11### IMPORTS
 12
 13import numpy as np
 14import spiceypy as sp
 15import matplotlib.pyplot as plt
 16import timeit
 17
 18from pyRTX.classes.Spacecraft import Spacecraft
 19from pyRTX.classes.Planet import Planet
 20from pyRTX.classes.Radiation import Albedo, Emissivity
 21from pyRTX.classes.Precompute import Precompute
 22from pyRTX.classes.LookUpTable import LookUpTable
 23from pyRTX.core.analysis_utils import epochRange2
 24
 25from numpy import floor, mod
 26
 27### ------------------------------------------------------------------------------------------------------- ###
 28### INPUTS
 29
 30ref_epc		=  "2010 may 10 09:25:00"
 31duration    =  10000  									  # seconds
 32timestep    =  100
 33METAKR      = '../example_data/LRO/metakernel_lro.tm'     # metakernel
 34obj_path    = '../example_data/LRO/'				      # folder with .obj files
 35lutfile     = 'luts/lro_accel_lut.nc'					  # lookup table file
 36base_flux   =  1361.5
 37ref_radius  =  1737.4
 38n_cores     =  10
 39sc_mass		=  2000  									  # can be a float, int or xarray [kg]
 40
 41### ------------------------------------------------------------------------------------------------------- ###
 42### OBJECTS DEFINITION
 43
 44# Time initialization
 45tic = timeit.default_timer()
 46
 47# Load the metakernel containing references to the necessary SPICE frames
 48sp.furnsh(METAKR)
 49
 50# Define epochs
 51epc_et0  = sp.str2et( ref_epc )
 52epc_et1  = epc_et0 + duration
 53epochs   = epochRange2(startEpoch = epc_et0, endEpoch = epc_et1, step = timestep)
 54 
 55# Define the Spacecraft Object (Refer to the class documentation for further details)
 56lro = Spacecraft( name = 'LRO',
 57                 
 58				  base_frame = 'LRO_SC_BUS', 					     # Name of the spacecraft body-fixed frame
 59      
 60                  mass = sc_mass,
 61      
 62				  spacecraft_model = {						         # Define a spacecraft model
 63                          
 64					'LRO_BUS': { 
 65							 'file' : obj_path + 'bus_rotated.obj',	 # .obj file of the spacecraft component
 66							 'frame_type': 'Spice',				     # type of frame (can be 'Spice' or 'UD'
 67							 'frame_name': 'LRO_SC_BUS',			 # Name of the frame
 68							 'center': [0.0,0.0,0.0],			     # Origin of the component
 69							 'diffuse': 0.1,				         # Diffuse reflect. coefficient
 70							 'specular': 0.3,				         # Specular reflect. coefficient
 71							 },
 72
 73					'LRO_SA': {	
 74							'file': obj_path + 'SA_recentred.obj',
 75							'frame_type': 'Spice',
 76							'frame_name': 'LRO_SA',
 77							'center': [-1,-1.1, -0.1],
 78							'diffuse': 0,
 79							'specular': 0.3,
 80							},
 81
 82
 83					'LRO_HGA': { 	
 84							'file': obj_path + 'HGA_recentred.obj',
 85							'frame_type': 'Spice',
 86							'frame_name': 'LRO_HGA',
 87							'center':[-0.99,    -0.3,  -3.1],
 88							'diffuse': 0.2,
 89							'specular': 0.1,
 90							},
 91					}
 92					)
 93
 94# Define the Moon object
 95moon = Planet(  fromFile = None,
 96                radius = ref_radius,
 97                name = 'Moon',
 98                bodyFrame = 'MOON_PA',
 99                sunFixedFrame = 'GSE_MOON',
100                units = 'km',
101                subdivs = 5,
102                )
103
104# Set the albedo and emissivity values
105# Here we use dummy values and assume that
106# albedo and emissivity are constant over the whole planet
107# and set a different dayside and nightside temperature
108# pyRTX supports also gridded values for albedo and emissivity (see examples/lro_alb_ir_grid.py)
109
110moon.albedo = 0.3
111moon.emissivity  = 0.8 
112moon.dayside_temperature = 300
113moon.nightside_temperature = 200
114
115# Load the Look up table
116LUT  = LookUpTable(lutfile)
117
118# Precomputation object
119prec = Precompute(epochs = epochs,)
120prec.precomputePlanetaryRadiation(lro, moon, LUT.moving_frames, correction='CN')
121prec.dump()
122
123# Create the albedo object
124albedo = Albedo(lro, LUT, moon, precomputation  = prec, baseflux  = base_flux,)
125
126# Create the thermal infrared object
127thermal_ir = Emissivity(lro, LUT, moon, precomputation  = prec, baseflux  = base_flux,)
128
129### ------------------------------------------------------------------------------------------------------- ###
130### COMPUTATIONS
131
132# Both the albedo and emissivity objects have a .compute() method.
133# This method returns the normalized fluxes, direction and albedo values
134# for each of the planet faces contributing to the computation
135# The general formula for computing the acceleration of an elementary face is:
136#
137# acc_i = L * albedo_value/mass * norm_flux
138#
139# where L is the normalized optical response of the spacecraft which can be computed
140# with raytracing setting a unitary radiance of the impacting rays
141# Due to the very high number of rays involved in these computations
142# it is mandatory to precompute L in the form of a lookup table.
143# Here we import the lookup table for LRO which can be computed using the example script
144# 'compute_lut.py'
145
146# Parallel computations
147alb_accel = albedo.compute(epochs, n_cores = n_cores)[0] * 1e3 
148ir_accel  = thermal_ir.compute(epochs, n_cores = n_cores)[0] * 1e3 
149
150# Always unload the SPICE kernels
151sp.unload(METAKR)
152
153### ... Elapsed time
154toc = timeit.default_timer()
155time_min = int(floor((toc-tic)/60))
156time_sec = int(mod((toc-tic), 60))
157print("")
158print("\t Elapsed time: %d min, %d sec" %(time_min, time_sec))
159print("")
160
161### ------------------------------------------------------------------------------------------------------- ###
162### PLOT
163
164epochs  = [float( epc - epc_et0 )/3600 for epc in epochs]
165
166# ALBEDO 
167fig, ax = plt.subplots(3, 1, figsize=(14,8), sharex = True)
168
169ax[0].plot(epochs, alb_accel[:,0], linewidth = 2, color = "tab:blue")
170ax[0].set_ylabel('X [m/s^2]')
171ax[1].plot(epochs, alb_accel[:,1], linewidth = 2, color = "tab:blue")
172ax[1].set_ylabel('Y [m/s^2]')
173ax[2].plot(epochs, alb_accel[:,2], linewidth = 2, color = "tab:blue")
174ax[2].set_ylabel('Z [m/s^2]')
175ax[2].set_xlabel('Hours from CA')
176fig.suptitle('Albedo in S/C body frame')
177plt.tight_layout()
178
179# IR
180fig, ax = plt.subplots(3, 1, figsize=(14,8), sharex = True)
181
182ax[0].plot(epochs, ir_accel[:,0], linewidth = 2, color = "tab:blue")
183ax[0].set_ylabel('X [m/s^2]')
184ax[1].plot(epochs, ir_accel[:,1], linewidth = 2, color = "tab:blue")
185ax[1].set_ylabel('Y [m/s^2]')
186ax[2].plot(epochs, ir_accel[:,2], linewidth = 2, color = "tab:blue")
187ax[2].set_ylabel('Z [m/s^2]')
188ax[2].set_xlabel('Hours from CA')
189fig.suptitle('IR in S/C body frame')
190plt.tight_layout()
191
192plt.show()
193
